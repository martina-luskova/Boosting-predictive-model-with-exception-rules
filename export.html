import re

def make_rule_matrix_with_conf(rule_texts, all_conditions, conf_list, id_list):
    binary_rows = []

    for rule, conf, rule_id in zip(rule_texts, conf_list, id_list):
        # split right and left side
        lhs, rhs = rule.split('=>')
        conditions_in_rule = [cond.strip() for cond in lhs.strip().split('&')]

        # extract target value - target(1) here
        match = re.search(r'target\((.*?)\)', rhs)
        target_val = int(match.group(1)) if match else None

        # create binary vector for conditions
        row = [1 if cond in conditions_in_rule else 0 for cond in all_conditions]

        # add target, conf and ID 
        row.extend([target_val, conf, rule_id])
        binary_rows.append(row)

    # final table
    columns = all_conditions + ['target', 'conf', 'rule_id']
    binary_df = pd.DataFrame(binary_rows, columns=columns)

    return binary_df

def apply_single_rule(test_df, conditions):
    mask = pd.Series([True] * len(test_df), index=test_df.index)

    for cond in conditions:
        col, val = cond[:-1].split('(')
        col = col.strip()
        val = val.strip()

        # if number - change type
        if test_df[col].dtype in [int, float]:
            try:
                val = float(val) if '.' in val else int(val)
            except ValueError:
                pass  # keep as str

        mask &= test_df[col] == val

    return mask
